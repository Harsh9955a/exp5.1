Mongoose Schema Definition Schema
const { Schema, model } = require('mongoose');

// ── Product Schema ──────────────────────────────────────────
const productSchema = new Schema({
  name: {
    type: String,
    required: [true, 'Product name is required'],
    trim: true,
    minlength: 2, maxlength: 100,
  },
  price: {
    type: Number,
    required: [true, 'Price is required'],
    min: [0, 'Price cannot be negative'],
  },
  category: {
    type: String,
    required: true,
    enum: {
      values: ['Electronics', 'Clothing', 'Appliances', 'Sports', 'Home'],
      message: '{VALUE} is not a valid category',
    },
  },
  description: { type: String, default: '' },
  stock: { type: Number, default: 0, min: 0 },
  inStock: { type: Boolean, default: true },
  brand: { type: String, default: '' },
  rating: { type: Number, default: null, min: 0, max: 5 },
  createdAt: { type: Date, default: Date.now },
  updatedAt: { type: Date, default: Date.now },
}, { timestamps: true });

// ── Middleware (pre-save hook) ───────────────────────────────
productSchema.pre('save', function(next) {
  this.inStock = this.stock > 0; // auto-set inStock
  this.updatedAt = new Date();
  next();
});

// ── Instance Method ─────────────────────────────────────────
productSchema.methods.applyDiscount = function(pct) {
  return this.price * (1 - pct / 100);
};

// ── Create Model ────────────────────────────────────────────
const Product = model('Product', productSchema);
module.exports = Product;
